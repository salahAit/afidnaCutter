    for (let i = 1; i < sorted.length; i++) {
        const last = chunks[chunks.length - 1];
        if (sorted[i].start <= last.end + gapThreshold) {
            last.end = Math.max(last.end, sorted[i].end);
        } else {
            chunks.push({ start: sorted[i].start, end: sorted[i].end });
        }
    }
    return chunks;
}
